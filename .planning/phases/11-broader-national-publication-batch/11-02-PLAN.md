export function getStateBySlug(slug: string) {
  const maybeStateEntry = contentGraph.states.find((entry) => entry.slug === slug);
  if (!maybeStateEntry) {
    return undefined;
  }

  return {
    ...maybeStateEntry,
    manifest: manifestBySlug.get(slug),
  };
}
